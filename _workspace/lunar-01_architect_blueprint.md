### 2.2 `src/lib/lunar-converter/conversion.ts`
Pure conversion engine (wraps injected `korean-lunar-calendar` lib):

```typescript
import { ConversionResult, ConversionError } from './schema';

// Engine injectable (for testing + future swap)
export interface LunarEngine {
  setSolarDate(y: number, m: number, d: number): boolean;
  getLunarCalendar(): { year: number; month: number; day: number; intercalation: boolean };
  setLunarDate(y: number, m: number, d: number, intercalation: boolean): boolean;
  getSolarCalendar(): { year: number; month: number; day: number };
  getKoreanGapja(): { year: string; month: string; day: string; intercalation: boolean };
}

const TABLE_YEAR_MIN = 1391;
const TABLE_YEAR_MAX = 2050;

export type ConversionInput = ConversionResult | ConversionError;

/**
 * Convert solar date → lunar date + sexagenary + zodiac
 * @param year 1391–2050
 * @param month 1–12
 * @param day 1–31
 * @param engine optional for testing (defaults to korean-lunar-calendar)
 * @returns ConversionResult | {error: 'out_of_range' | 'invalid_date'}
 */
export async function solarToLunar(
  year: number,
  month: number,
  day: number,
  engine?: LunarEngine
): Promise<ConversionResult | ConversionError> {
  const eng = engine || (await loadEngine());
  
  // Range validation
  if (year < TABLE_YEAR_MIN || year > TABLE_YEAR_MAX) {
    return { error: 'out_of_range' };
  }
  
  // Date validation (month 1–12, day per month)
  if (month < 1 || month > 12) return { error: 'invalid_date' };
  const maxDay = daysInMonth(month, year);
  if (day < 1 || day > maxDay) return { error: 'invalid_date' };
  
  // Engine call
  if (!eng.setSolarDate(year, month, day)) {
    return { error: 'invalid_date' };
  }
  
  const lunar = eng.getLunarCalendar();
  const gapja = eng.getKoreanGapja();
  
  const sex = computeSexagenary(lunar.year);
  const zod = computeZodiac(lunar.year);
  
  return {
    solarDate: { year, month, day },
    lunarDate: {
      year: lunar.year,
      month: lunar.month,
      day: lunar.day,
      isLeap: lunar.intercalation,
    },
    sexagenary: {
      name: sex.name,
      hanja: sex.hanja,
      english: sex.english,
      stemIndex: sex.stemIndex,
      branchIndex: sex.branchIndex,
    },
    zodiac: {
      key: zod.key,
      emoji: zod.emoji,
      branchIndex: zod.branchIndex,
    },
  };
}

/**
 * Convert lunar date → solar date + sexagenary + zodiac
 * @param year 1391–2050
 * @param month 1–12
 * @param day 1–30
 * @param isLeap true if leap month
 * @param engine optional for testing
 * @returns ConversionResult | {error: 'out_of_range' | 'no_leap_month' | 'invalid_date'}
 */
export async function lunarToSolar(
  year: number,
  month: number,
  day: number,
  isLeap: boolean,
  engine?: LunarEngine
): Promise<ConversionResult | ConversionError> {
  const eng = engine || (await loadEngine());
  
  // Range validation
  if (year < TABLE_YEAR_MIN || year > TABLE_YEAR_MAX) {
    return { error: 'out_of_range' };
  }
  
  // Month/day validation (defer to engine)
  if (month < 1 || month > 12 || day < 1 || day > 30) {
    return { error: 'invalid_date' };
  }
  
  // Engine call (returns false if leap month doesn't exist or date invalid)
  if (!eng.setLunarDate(year, month, day, isLeap)) {
    // Distinguish: if isLeap and false, it's likely no_leap_month
    // Otherwise invalid_date
    return isLeap ? { error: 'no_leap_month' } : { error: 'invalid_date' };
  }
  
  const solar = eng.getSolarCalendar();
  const gapja = eng.getKoreanGapja();
  
  const sex = computeSexagenary(year);
  const zod = computeZodiac(year);
  
  return {
    solarDate: { year: solar.year, month: solar.month, day: solar.day },
    lunarDate: { year, month, day, isLeap },
    sexagenary: {
      name: sex.name,
      hanja: sex.hanja,
      english: sex.english,
      stemIndex: sex.stemIndex,
      branchIndex: sex.branchIndex,
    },
    zodiac: {
      key: zod.key,
      emoji: zod.emoji,
      branchIndex: zod.branchIndex,
    },
  };
}

// INTERNAL: Load engine (dynamic import, code-split)
async function loadEngine(): Promise<LunarEngine> {
  const mod = await import('korean-lunar-calendar');
  return {
    setSolarDate: (y, m, d) => mod.setSolarDate?.(y, m, d) ?? false,
    getLunarCalendar: () => mod.getLunarCalendar?.() ?? { year: 0, month: 0, day: 0, intercalation: false },
    setLunarDate: (y, m, d, i) => mod.setLunarDate?.(y, m, d, i) ?? false,
    getSolarCalendar: () => mod.getSolarCalendar?.() ?? { year: 0, month: 0, day: 0 },
    getKoreanGapja: () => mod.getKoreanGapja?.() ?? { year: '', month: '', day: '', intercalation: false },
  };
}

// INTERNAL: Days per month (Gregorian)
function daysInMonth(m: number, y: number): number {
  const days = [31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31];
  if (m === 2 && isLeapYear(y)) return 29;
  return days[m - 1] ?? 31;
}

function isLeapYear(y: number): boolean {
  return (y % 4 === 0 && y % 100 !== 0) || y % 400 === 0;
}

import { computeZodiac } from './zodiac';
```

### 2.3 `src/lib/lunar-converter/sexagenary.ts`

```typescript
/**
 * Sexagenary cycle: 10 stems (天干) × 12 branches (地支) = 60 years
 * Anchor: year 4 CE = 甲子 (1,1) in the cycle
 * Formula: stemIndex = (year - 4) % 10; branchIndex = (year - 4) % 12
 */

export interface SexagenaryInfo {
  name: string; // e.g., "갑진"
  hanja: string; // e.g., "甲辰"
  english: string; // e.g., "Wood Dragon"
  stemIndex: number; // 0–9
  branchIndex: number; // 0–11
}

// 10 heavenly stems
const STEMS_KO = ['갑', '을', '병', '정', '무', '기', '경', '신', '임', '계'];
const STEMS_HANJA = ['甲', '乙', '丙', '丁', '戊', '己', '庚', '辛', '壬', '癸'];
const STEMS_ELEMENT = ['wood', 'wood', 'fire', 'fire', 'earth', 'earth', 'metal', 'metal', 'water', 'water'];
const STEMS_YIN_YANG = ['yang', 'yin', 'yang', 'yin', 'yang', 'yin', 'yang', 'yin', 'yang', 'yin'];

// 12 earthly branches
const BRANCHES_KO = ['쥐', '소', '호랑이', '토끼', '용', '뱀', '말', '양', '원숭이', '닭', '개', '돼지'];
const BRANCHES_HANJA = ['子', '丑', '寅', '卯', '辰', '巳', '午', '未', '申', '酉', '戌', '亥'];
const BRANCHES_ANIMAL = ['rat', 'ox', 'tiger', 'rabbit', 'dragon', 'snake', 'horse', 'goat', 'monkey', 'rooster', 'dog', 'pig'];
const BRANCHES_ELEMENT = ['water', 'earth', 'wood', 'wood', 'earth', 'fire', 'fire', 'earth', 'metal', 'metal', 'water', 'water'];
const BRANCHES_EMOJI = ['🐭', '🐂', '🐅', '🐰', '🐉', '🐍', '🐴', '🐑', '🐵', '🐔', '🐕', '🐷'];

/**
 * Compute sexagenary info for a lunar year
 * @param lunarYear any year (formula works for negatives, but in practice 1391–2050)
 * @returns SexagenaryInfo with name (korean), hanja, english (element + animal)
 */
export function computeSexagenary(lunarYear: number): SexagenaryInfo {
  const stemIndex = ((lunarYear - 4) % 10 + 10) % 10; // Ensure positive mod
  const branchIndex = ((lunarYear - 4) % 12 + 12) % 12;
  
  const stemKo = STEMS_KO[stemIndex];
  const branchKo = BRANCHES_KO[branchIndex];
  const name = `${stemKo}${branchKo}`;
  
  const stemHanja = STEMS_HANJA[stemIndex];
  const branchHanja = BRANCHES_HANJA[branchIndex];
  const hanja = `${stemHanja}${branchHanja}`;
  
  // English: "Wood Dragon", "Metal Rat", etc.
  const stemElement = STEMS_ELEMENT[stemIndex];
  const branchAnimal = BRANCHES_ANIMAL[branchIndex];
  const english = `${capitalize(stemElement)} ${capitalize(branchAnimal)}`;
  
  return {
    name, // "갑진"
    hanja, // "甲辰"
    english, // "Wood Dragon"
    stemIndex,
    branchIndex,
  };
}

/**
 * Get stem info by index (0–9)
 */
export function getStemInfo(stemIndex: number) {
  return {
    ko: STEMS_KO[stemIndex],
    hanja: STEMS_HANJA[stemIndex],
    element: STEMS_ELEMENT[stemIndex],
    yinYang: STEMS_YIN_YANG[stemIndex],
  };
}

/**
 * Get branch info by index (0–11)
 */
export function getBranchInfo(branchIndex: number) {
  return {
    ko: BRANCHES_KO[branchIndex],
    hanja: BRANCHES_HANJA[branchIndex],
    animal: BRANCHES_ANIMAL[branchIndex],
    element: BRANCHES_ELEMENT[branchIndex],
    emoji: BRANCHES_EMOJI[branchIndex],
  };
}

// INTERNAL
function capitalize(s: string): string {
  return s.charAt(0).toUpperCase() + s.slice(1);
}
```

### 2.5 `src/lib/lunar-converter/recents.ts`
Immutable localStorage operations:

```typescript
import { z } from 'zod';
import { Recent, StoredRecents, storedRecentsSchema } from './schema';

const STORE_VERSION = 1;
const RECENTS_MAX = 10;
const STORE_KEY = 'jurepi-lunar-converter';

/**
 * Push a new recent conversion (solar + lunar).
 * Deduplicates (move existing to front if present).
 * Returns new entries array (max 10 MRU first).
 */
export function pushRecent(
  current: Recent[],
  solar: string, // YYYY-MM-DD
  lunar: string // YYYY-MM-DD or YYYY-윤MM-DD
): Recent[] {
  const now = Date.now();
  const newEntry: Recent = {
    solarDate: solar,
    lunarDate: lunar,
    ts: now,
  };
  
  // Remove if already exists (same solar + lunar pair)
  const filtered = current.filter(
    (r) => !(r.solarDate === solar && r.lunarDate === lunar)
  );
  
  // Prepend new, trim to max
  return [newEntry, ...filtered].slice(0, RECENTS_MAX);
}

/**
 * Prune entries with dates outside table bounds (1391–2050).
 */
export function pruneUnknown(entries: Recent[]): Recent[] {
  return entries.filter((r) => {
    const [y] = r.solarDate.split('-');
    const year = parseInt(y, 10);
    return year >= 1391 && year <= 2050;
  });
}

/**
 * Load recents from localStorage, parse + prune, return entries.
 * If corrupt/missing, return [].
 */
export function loadRecents(): Recent[] {
  try {
    const raw = localStorage.getItem(STORE_KEY);
    if (!raw) return [];
    
    const parsed = JSON.parse(raw);
    const validated = storedRecentsSchema.parse(parsed);
    const pruned = pruneUnknown(validated.entries);
    
    return pruned;
  } catch {
    // Corrupt or missing → start fresh
    return [];
  }
}

/**
 * Save recents to localStorage.
 * Silently fails if quota exceeded or unavailable.
 */
export function saveRecents(entries: Recent[]): void {
  try {
    const blob: StoredRecents = { version: STORE_VERSION, entries };
    localStorage.setItem(STORE_KEY, JSON.stringify(blob));
  } catch {
    // Quota full or unavailable → keep in-memory only
  }
}

/**
 * Serialize recent for display.
 */
export function serializeRecent(r: Recent): string {
  return `${r.solarDate} → ${r.lunarDate}`;
}
```

## 3. Sexagenary & Zodiac Mathematical Anchors

**Anchor: Year 4 CE = 甲子 (1,1) — the "0th" sexagenary year in traditional reckoning.**

### Sexagenary Formula:
- **Stem Index** (天干, 0–9): `(year - 4) % 10` → {甲 을 丙 丁 戊 己 庚 辛 壬 癸}
- **Branch Index** (地支, 0–11): `(year - 4) % 12` → {子 丑 寅 卯 辰 巳 午 未 申 酉 戌 亥}
- **Full Name** (간지, 60-year cycle): concatenate stem + branch hanja.
- **English**: stem element + branch animal, e.g., "Wood Dragon" (甲辰).

### Zodiac Formula:
- **Branch Index** = `(year - 4) % 12`
- **Animal**: {rat, ox, tiger, rabbit, dragon, snake, horse, goat, monkey, rooster, dog, pig}

```
12. lunarToSolar(2024, 4, 1, isLeap=TRUE) → error 'no_leap_month'   (2024 has NO leap month)
    - (2023 has 윤2월, 2020 윤4월, 2025 윤6월 — use these for leap positives)
13. solarToLunar(2024, 2, 31) → error 'invalid_date'
```

간지 sanity (formula matches lib exactly, verified):
- 2024: (2024-4)%10=0→甲, %12=4→辰 ⇒ 甲辰/용 ✓
- 2023: (2023-4)%10=9→癸, %12=3→卯 ⇒ 癸卯/토끼 ✓
- 1390: (1390-4)%10=6→庚, %12=6→午 ⇒ 庚午/말 ✓

## 4. useConverter Hook Contract

**Location:** `src/components/tools/lunar-converter/useConverter.ts`

```typescript
/**
 * Manages converter state: solar/lunar date inputs, conversion result, recent history.
 * Owns localStorage + copy adapter.
 * 
 * This hook is the USE-CASE layer: translates UI inputs (which calendar changed?)
 * into domain calls (solarToLunar vs lunarToSolar).
 */

import { useCallback, useEffect, useReducer } from 'react';
import { solarToLunar, lunarToSolar, type LunarEngine } from '@/lib/lunar-converter/conversion';
import { loadRecents, saveRecents, pushRecent } from '@/lib/lunar-converter/recents';
import type { ConversionResult, ConversionError, Recent } from '@/lib/lunar-converter/schema';

export interface ConverterState {
  // Current input (which calendar did user edit?)
  solarYear: number;
  solarMonth: number;
  solarDay: number;
  lunarYear: number;
  lunarMonth: number;
  lunarDay: number;
  lunarIsLeap: boolean;
  
  // Last conversion result
  result: ConversionResult | ConversionError | null;
  
  // Recent history
  recents: Recent[];
  
  // Copy toast state
  copyKey: 'solar' | 'lunar' | 'both' | null; // Which was last copied
  
  // Loading state (engine dynamic import)
  isLoading: boolean;
  isMounted: boolean;
}

export type ConverterAction =
  | { type: 'SET_SOLAR'; year: number; month: number; day: number }
  | { type: 'SET_LUNAR'; year: number; month: number; day: number; isLeap: boolean }
  | { type: 'SET_TODAY'; solarYear: number; solarMonth: number; solarDay: number; lunarYear: number; lunarMonth: number; lunarDay: number; lunarIsLeap: boolean }
  | { type: 'SET_RESULT'; result: ConversionResult | ConversionError }
  | { type: 'SET_RECENTS'; recents: Recent[] }
  | { type: 'ADD_RECENT'; recent: Recent }
  | { type: 'SET_COPY_KEY'; key: ConverterState['copyKey'] }
  | { type: 'SET_MOUNTED'; mounted: boolean };

/**
 * useConverter() hook
 * @param engine optional LunarEngine for testing; defaults to dynamic import of korean-lunar-calendar
 * @returns current state + dispatch actions
 */
export function useConverter(engine?: LunarEngine) {
  const [state, dispatch] = useReducer(converterReducer, initialState);
  
  // Load recents on mount
  useEffect(() => {
    const recents = loadRecents();
    dispatch({ type: 'SET_RECENTS', recents });
    dispatch({ type: 'SET_MOUNTED', mounted: true });
  }, []);
  
  // Conversion effect: when solar/lunar/isLeap changes, convert
  useEffect(() => {
    if (!state.isMounted) return;
    
    // Determine which direction (heuristic: assume last non-zero input changed)
    // For simplicity, prioritize solar over lunar (solar input = trigger solar→lunar)
    // This is a UX heuristic; in real impl, track which input changed via form event
    
    const convert = async () => {
      if (state.solarYear && state.solarMonth && state.solarDay) {
        const result = await solarToLunar(state.solarYear, state.solarMonth, state.solarDay, engine);
        dispatch({ type: 'SET_RESULT', result });
        
        if ('solarDate' in result) {
          // Successful: update lunar inputs to match result
          // (Avoid ping-pong: only update if different)
          if (
            state.lunarYear !== result.lunarDate.year ||
            state.lunarMonth !== result.lunarDate.month ||
            state.lunarDay !== result.lunarDate.day ||
            state.lunarIsLeap !== result.lunarDate.isLeap
          ) {
            dispatch({
              type: 'SET_LUNAR',
              year: result.lunarDate.year,
              month: result.lunarDate.month,
              day: result.lunarDate.day,
              isLeap: result.lunarDate.isLeap,
            });
          }
        }
      }
    };
    
    convert();
  }, [state.solarYear, state.solarMonth, state.solarDay, state.isMounted, engine]);
  
  // Similarly for lunar → solar (alternate path)
  // ...
  
  // Copy to clipboard
  const handleCopy = useCallback(async (key: 'solar' | 'lunar' | 'both') => {
    if (!('solarDate' in state.result!)) return;
    
    const res = state.result as ConversionResult;
    const solarStr = `${res.solarDate.year}-${String(res.solarDate.month).padStart(2, '0')}-${String(res.solarDate.day).padStart(2, '0')}`;
    const lunarStr = `${res.lunarDate.year}-${String(res.lunarDate.month).padStart(2, '0')}-${String(res.lunarDate.day).padStart(2, '0')}${res.lunarDate.isLeap ? '(윤)' : ''}`;
    
    const text =
      key === 'solar'
        ? solarStr
        : key === 'lunar'
        ? lunarStr
        : `${solarStr} / ${lunarStr}`;
    
    try {
      await navigator.clipboard.writeText(text);
      dispatch({ type: 'SET_COPY_KEY', key });
      setTimeout(() => dispatch({ type: 'SET_COPY_KEY', key: null }), 1500);
    } catch {
      // Silent fail
    }
  }, [state.result]);
  
  // Load a recent conversion
  const loadRecent = useCallback((r: Recent) => {
    // Parse recent dates and set inputs
    const [sy, sm, sd] = r.solarDate.split('-').map(Number);
    const lunarPart = r.lunarDate.replace('(윤)', '');
    const [ly, lm, ld] = lunarPart.split('-').map(Number);
    const isLeap = r.lunarDate.includes('(윤)');
    
    dispatch({ type: 'SET_SOLAR', year: sy, month: sm, day: sd });
    dispatch({ type: 'SET_LUNAR', year: ly, month: lm, day: ld, isLeap });
  }, []);
  
  return {
    ...state,
    setSolar: (y: number, m: number, d: number) => dispatch({ type: 'SET_SOLAR', year: y, month: m, day: d }),
    setLunar: (y: number, m: number, d: number, isLeap: boolean) => dispatch({ type: 'SET_LUNAR', year: y, month: m, day: d, isLeap }),
    setToday: (sy: number, sm: number, sd: number, ly: number, lm: number, ld: number, isLeap: boolean) =>
      dispatch({ type: 'SET_TODAY', solarYear: sy, solarMonth: sm, solarDay: sd, lunarYear: ly, lunarMonth: lm, lunarDay: ld, lunarIsLeap: isLeap }),
    copy: handleCopy,
    loadRecent,
  };
}

const initialState: ConverterState = {
  solarYear: new Date().getFullYear(),
  solarMonth: new Date().getMonth() + 1,
  solarDay: new Date().getDate(),
  lunarYear: 0,
  lunarMonth: 0,
  lunarDay: 0,
  lunarIsLeap: false,
  result: null,
  recents: [],
  copyKey: null,
  isLoading: false,
  isMounted: false,
};

function converterReducer(state: ConverterState, action: ConverterAction): ConverterState {
  switch (action.type) {
    case 'SET_SOLAR':
      return { ...state, solarYear: action.year, solarMonth: action.month, solarDay: action.day };
    case 'SET_LUNAR':
      return { ...state, lunarYear: action.year, lunarMonth: action.month, lunarDay: action.day, lunarIsLeap: action.isLeap };
    case 'SET_RESULT':
      return { ...state, result: action.result };
    case 'SET_RECENTS':
      return { ...state, recents: action.recents };
    case 'ADD_RECENT':
      // On successful conversion, persist
      const newRecents = pushRecent(state.recents, `${state.solarYear}-...`, `${state.lunarYear}-...`);
      saveRecents(newRecents);
      return { ...state, recents: newRecents };
    case 'SET_COPY_KEY':
      return { ...state, copyKey: action.key };
    case 'SET_MOUNTED':
      return { ...state, isMounted: action.mounted };
    default:
      return state;
  }
}
```

## 8. Invariants & Risks

### Invariants (Must-Hold):
- **Range Bound:** All conversions strictly within [1391, 2050] (KASI official).
- **Sexagenary Math:** (year - 4) % 10 = stem, (year - 4) % 12 = branch. Anchor = year 4 CE = 甲子.
- **Leap Month Uniqueness:** Each lunar year has ≤1 leap month; if none, toggle must error 'no_leap_month'.
- **Locale Binding:** useLocale() → Intl APIs (never i18n keys to Intl).
- **localStorage Resilience:** Corrupt blob → start fresh (no throw); quota full → keep in-memory.
- **Immutability:** recents.ts returns new arrays (no mutation).
- **SEO SSR:** Intro/HowTo/FAQ/JSON-LD rendered outside mounted gate (AI crawlers don't execute JS).

### Risks & Mitigations:

| Risk | Mitigation |
|------|-----------|
| korean-lunar-calendar API mismatch | Verify `package.json` post-install; domain-engineer checks actual API (setSolarDate, getLunarCalendar, etc.) and wraps accordingly. |
| Leap month edge case | Domain tests must assert: year with leap month (e.g., 2023년 윤2월), year without (e.g., 2024), toggle on both scenarios. |
